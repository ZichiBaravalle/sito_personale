Questo qua sotto invece è un simpatico e complicato programma in C con il goto che crea un albero in console inserendo il numero della grandezza con la variabile. 



void DisegnaAlberoGOTO(int grandezza){
    int cont = 1, spazi = grandezza - 1, livelli = 0;
    int stelle = 1;
    loop_principale:
    if(cont != grandezza)
    {
        stelle = 0;
        seconda_stella:
        if(stelle == 1)
        {
            spazi = ((livelli - 1) * 2) + 1;
        }
        else
            spazi = grandezza - cont;
        spazi:
        if (spazi > 0 && stelle == 0) {
            printf(" ");
            spazi--;
            goto spazi;
        }
        spazi2:
        if(spazi > 0 && stelle == 1)
        {
            printf(" ");
            spazi--;
            goto spazi2;
        }
        else
        {
            printf("*");
            stelle++;
        }
        if(cont > 1 && stelle == 1)
            livelli++;
        if(livelli == 0 || stelle > 1)
        {
            printf("\n");
            cont++;
            goto loop_principale;
        }
        else
            goto seconda_stella;
    }
    cont = 0;
    fondo:
    if (cont < (grandezza * 2) - 1) {
        printf("*");
        cont++;
        goto fondo;
    }
}
